Apps Script Code
Paste this into script.google.com → New project → Deploy as Web App
function doPost(e) { var sheetId = 'YOUR_SHEET_ID_HERE'; var ss = SpreadsheetApp.openById(sheetId); var sheet = ss.getSheetByName('Scans') || ss.getActiveSheet(); // Read from form parameter (URLSearchParams) or postData var raw = '{}'; if (e.parameter && e.parameter.data) { raw = e.parameter.data; } else if (e.postData && e.postData.contents) { raw = e.postData.contents; } var data = JSON.parse(raw); if (sheet.getLastRow() === 0) { sheet.appendRow(['ScanID','Model','RawData','Date','Time','Status']); } sheet.appendRow([ data.scanId, data.model, data.rawData, data.date, data.time, data.status || 'OK' ]); return ContentService .createTextResponse(JSON.stringify({result:'OK'})) .setMimeType(ContentService.MimeType.JSON); }
Replace YOUR_SHEET_ID_HERE with the ID from your Sheet URL. Deploy → Execute as: Me → Who has access: Anyone.
